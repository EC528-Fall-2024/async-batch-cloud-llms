
## Deploy 
#The folowing command deploys the function from the terminal
gcloud functions deploy batchprocessor \
  --gen2 \
  --region=us-central1 \
  --runtime=python312 \
  --timeout=300 \
  --source=. \
  --entry-point=go \
  --trigger-topic=StartJob \
  --allow-unauthenticated \
  --verbosity=debug
